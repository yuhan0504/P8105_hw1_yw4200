library(ggplot2)

# Make a scatterplot of temp (y) vs time_hour (x)
scatterplot <- ggplot(
  early_january_weather, 
  aes(x = time_hour, y = temp, color = humid)) +
  geom_point() +
  labs(x = "Time Hour", y = "Temperature", color = "Humidity")

print(scatterplot)

# Export scatterplot
ggsave("scatterplot of temp vs time_hour.png", plot =
         scatterplot)
